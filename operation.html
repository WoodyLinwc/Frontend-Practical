Array Operations

Creation and Basic Operations

// Creating arrays
const arr1 = [1, 2, 3];
const arr2 = new Array(3);          // [empty Ã— 3]
const arr3 = Array.of(1, 2, 3);     // [1, 2, 3]
const arr4 = Array.from({length: 3}, (_, i) => i); // [0, 1, 2]

// Adding/removing elements
arr.push(4);           // add to end, returns new length
arr.pop();             // remove from end, returns removed element
arr.unshift(0);        // add to beginning, returns new length
arr.shift();           // remove from beginning, returns removed element
arr.splice(1, 1, 'new'); // remove 1 element at index 1, insert 'new'

Array Searching and Checking

const arr = [1, 2, 3, 4, 5];
arr.indexOf(3);        // 2 (first occurrence)
arr.lastIndexOf(3);    // 2 (last occurrence)
arr.includes(3);       // true
arr.find(x => x > 3);  // 4 (first match)
arr.findIndex(x => x > 3); // 3 (index of first match)
arr.some(x => x > 4);  // true (at least one matches)
arr.every(x => x > 0); // true (all match)

Array Transformation

const arr = [1, 2, 3, 4, 5];
arr.map(x => x * 2);                    // [2, 4, 6, 8, 10]
arr.filter(x => x > 2);                 // [3, 4, 5]
arr.reduce((sum, x) => sum + x, 0);     // 15
arr.reduceRight((sum, x) => sum + x, 0); // 15 (right to left)
arr.flatMap(x => [x, x * 2]);          // [1, 2, 2, 4, 3, 6, 4, 8, 5, 10]

Array Sorting and Reversing

const arr = [3, 1, 4, 1, 5];
arr.sort();                    // [1, 1, 3, 4, 5] (mutates original)
arr.sort((a, b) => b - a);     // [5, 4, 3, 1, 1] (descending)
arr.reverse();                 // [5, 1, 4, 1, 3] (mutates original)
[...arr].sort();               // non-mutating sort

Array Slicing and Joining

const arr = [1, 2, 3, 4, 5];
arr.slice(1, 3);       // [2, 3] (from index 1 to 3, exclusive)
arr.slice(-2);         // [4, 5] (last 2 elements)
arr.join('-');         // "1-2-3-4-5"
arr.concat([6, 7]);    // [1, 2, 3, 4, 5, 6, 7]

Array Flattening and Copying

// Flattening arrays
const nested = [1, [2, 3], [4, [5, 6]]];
const flat1 = nested.flat();        // [1, 2, 3, 4, [5, 6]] - one level
const flatAll = nested.flat(Infinity); // [1, 2, 3, 4, 5, 6] - all levels

// Shallow copying arrays
const arr = [1, 2, 3];
const copy1 = [...arr];           // spread operator
const copy2 = Array.from(arr);    // Array.from()
const copy3 = arr.slice();        // slice()

// Converting array to Set (removes duplicates)
const arr = [1, 2, 2, 3, 3, 4];
const uniqueSet = new Set(arr);        // Set {1, 2, 3, 4}
const uniqueArray = [...new Set(arr)]; // [1, 2, 3, 4]

Object Operations

Object Creation

const obj1 = {a: 1, b: 2};
const obj2 = new Object();
const obj3 = Object.create(null);           // no prototype
const obj4 = Object.create(obj1);           // inherits from obj1
const obj5 = Object.fromEntries([['a', 1], ['b', 2]]); // from entries

Object Property Operations

const obj = {a: 1, b: 2};
obj.hasOwnProperty('a');         // true
'a' in obj;                      // true
obj.propertyIsEnumerable('a');   // true
Object.getOwnPropertyNames(obj); // ['a', 'b']
Object.getOwnPropertyDescriptor(obj, 'a'); // {value: 1, writable: true, ...}

Object Manipulation

const obj1 = {a: 1, b: 2};
const obj2 = {c: 3, d: 4};

// Copying and merging
const copy = {...obj1};
const merged = {...obj1, ...obj2};
const assigned = Object.assign({}, obj1, obj2);

// Freezing and sealing
Object.freeze(obj);    // prevents all changes
Object.seal(obj);      // prevents adding/removing properties
Object.preventExtensions(obj); // prevents adding properties

Object Iteration

const obj = {a: 1, b: 2, c: 3};

// Getting object keys/values/entries
const keys = Object.keys(obj);       // ['a', 'b', 'c']
const values = Object.values(obj);   // [1, 2, 3]
const entries = Object.entries(obj); // [['a', 1], ['b', 2], ['c', 3]]

// Iteration methods
for (const key in obj) { /* iterate keys */ }
for (const [key, value] of Object.entries(obj)) { /* iterate entries */ }
Object.keys(obj).forEach(key => { /* iterate keys */ });
Object.values(obj).forEach(value => { /* iterate values */ });

String Operations

String Creation and Basic Operations

const str1 = "hello";
const str2 = String(123);
const str3 = `template ${str1}`;
const str4 = str1.repeat(3);         // "hellohellohello"
const str5 = str1.padStart(10, '0'); // "00000hello"
const str6 = str1.padEnd(10, '!');   // "hello!!!!!"

String Searching

const str = "hello world";
str.indexOf('o');          // 4 (first occurrence)
str.lastIndexOf('o');      // 7 (last occurrence)
str.includes('world');     // true
str.startsWith('hello');   // true
str.endsWith('world');     // true
str.search(/w\w+/);        // 6 (regex search)
str.match(/l/g);           // ['l', 'l', 'l'] (all matches)

String Transformation

const str = "  Hello World  ";
str.trim();                // "Hello World"
str.trimStart();           // "Hello World  "
str.trimEnd();             // "  Hello World"
str.toLowerCase();         // "  hello world  "
str.toUpperCase();         // "  HELLO WORLD  "
str.replace('World', 'JS'); // "  Hello JS  "
str.replaceAll('l', 'L');  // "  HeLLo WorLd  "

// Converting cases
const str = "Hello World";
str.toLowerCase();     // "hello world"
str.toUpperCase();     // "HELLO WORLD"
str.charAt(0).toUpperCase() + str.slice(1); // "Hello World" (capitalize first)

String Splitting and Slicing

const str = "a,b,c,d";
str.split(',');            // ['a', 'b', 'c', 'd']
str.split('');             // ['a', ',', 'b', ',', 'c', ',', 'd']
str.slice(2, 5);           // "b,c"
str.substring(2, 5);       // "b,c"
str.substr(2, 3);          // "b,c" (deprecated)
str.charAt(0);             // "a"
str.charCodeAt(0);         // 97

// String to array and back
const str = "hello";
const arr = str.split('');      // ['h', 'e', 'l', 'l', 'o']
const back = arr.join('');      // "hello"
const words = "a b c".split(' '); // ['a', 'b', 'c']

Number Operations

Number Creation and Checking

const num = 42;
Number.isInteger(42);      // true
Number.isFinite(42);       // true
Number.isNaN(NaN);         // true
Number.isSafeInteger(42);  // true
Number.parseFloat('3.14'); // 3.14
Number.parseInt('42px');   // 42

Number Formatting

const num = 3.14159;
num.toFixed(2);            // "3.14"
num.toPrecision(3);        // "3.14"
num.toExponential(2);      // "3.14e+0"
num.toString(16);          // "3.243f6a8885a3" (hex)

Math Operations

Math.abs(-5);              // 5
Math.round(3.7);           // 4
Math.floor(3.7);           // 3
Math.ceil(3.2);            // 4
Math.min(1, 2, 3);         // 1
Math.max(1, 2, 3);         // 3
Math.random();             // random number 0-1
Math.pow(2, 3);            // 8
Math.sqrt(16);             // 4

Date Operations

Date Creation

const now = new Date();
const specific = new Date('2023-12-25');
const fromTimestamp = new Date(1640995200000);
const fromParts = new Date(2023, 11, 25); // month is 0-indexed

Date Methods

const date = new Date();
date.getTime();            // timestamp
date.getFullYear();        // 2023
date.getMonth();           // 11 (0-indexed)
date.getDate();            // 25
date.getDay();             // 1 (0=Sunday)
date.getHours();           // 14
date.toISOString();        // "2023-12-25T14:30:00.000Z"
date.toLocaleDateString(); // "12/25/2023"

Set Operations

Set Creation and Methods

const set = new Set([1, 2, 3, 3]); // {1, 2, 3}
set.add(4);                // add element
set.has(3);                // true
set.delete(2);             // remove element
set.clear();               // remove all
set.size;                  // number of elements
Array.from(set);           // convert to array
[...set];                  // spread to array

Set Operations

const set1 = new Set([1, 2, 3]);
const set2 = new Set([3, 4, 5]);

// Union
const union = new Set([...set1, ...set2]); // {1, 2, 3, 4, 5}

// Intersection
const intersection = new Set([...set1].filter(x => set2.has(x))); // {3}

// Difference
const difference = new Set([...set1].filter(x => !set2.has(x))); // {1, 2}

Map Operations

Map Creation and Methods

const map = new Map();
map.set('key1', 'value1');
map.set('key2', 'value2');
map.get('key1');           // 'value1'
map.has('key1');           // true
map.delete('key1');        // remove entry
map.clear();               // remove all
map.size;                  // number of entries

// From array of arrays
const map2 = new Map([['a', 1], ['b', 2]]);

Map Iteration

const map = new Map([['a', 1], ['b', 2]]);
for (const [key, value] of map) { /* iterate entries */ }
for (const key of map.keys()) { /* iterate keys */ }
for (const value of map.values()) { /* iterate values */ }
map.forEach((value, key) => { /* iterate */ });

RegExp Creation and Methods

const str = "Hello World";
regex1.test(str);          // true

Async Operations

Promises

const promise = new Promise((resolve, reject) => {
  setTimeout(() => resolve('done'), 1000);
});

promise.then(result => console.log(result));
promise.catch(error => console.error(error));
promise.finally(() => console.log('cleanup'));

// Promise utilities
Promise.all([promise1, promise2]);     // wait for all
Promise.allSettled([promise1, promise2]); // wait for all, don't fail fast
Promise.race([promise1, promise2]);    // first to resolve/reject
Promise.resolve(value);                // resolved promise
Promise.reject(error);                 // rejected promise

Async/Await

async function fetchData() {
  try {
    const result = await fetch('https://api.example.com');
    return await result.json();
  } catch (error) {
    console.error(error);
  }
}

Destructuring

Array Destructuring

const arr = [1, 2, 3, 4, 5];
const [first, second, ...rest] = arr;  // first=1, second=2, rest=[3,4,5]
const [a, , c] = arr;                  // skip elements
const [x = 0, y = 0] = [1];            // default values

Object Destructuring

const obj = {a: 1, b: 2, c: 3};
const {a, b} = obj;                    // a=1, b=2
const {a: x, b: y} = obj;              // rename: x=1, y=2
const {a, b, d = 4} = obj;             // default values
const {a, ...rest} = obj;              // rest={b:2, c:3}

Utility Functions

Common Utility Patterns

// Deep clone (simple objects)
const deepClone = obj => JSON.parse(JSON.stringify(obj));

// Remove duplicates
const unique = arr => [...new Set(arr)];

// Chunk array
const chunk = (arr, size) => 
  Array.from({length: Math.ceil(arr.length / size)}, (_, i) =>
    arr.slice(i * size, i * size + size)
  );

// Debounce
const debounce = (func, delay) => {
  let timeoutId;
  return (...args) => {
    clearTimeout(timeoutId);
    timeoutId = setTimeout(() => func(...args), delay);
  };
};

// Throttle
const throttle = (func, delay) => {
  let lastCall = 0;
  return (...args) => {
    const now = Date.now();
    if (now - lastCall >= delay) {
      lastCall = now;
      func(...args);
    }
  };
};

// Removing Falsy Values
const arr = [0, 1, false, 2, "", 3, null, undefined, 4];
const truthy = arr.filter(Boolean); // [1, 2, 3, 4]

Working with JSON

const obj = {name: "John", age: 30};
const json = JSON.stringify(obj);    // '{"name":"John","age":30}'
const parsed = JSON.parse(json);     // {name: "John", age: 30}
